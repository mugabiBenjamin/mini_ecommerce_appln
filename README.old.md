```sh
# Create a new Flutter project named mini_ecommerce_appln
flutter create mini_ecommerce_appln
cd mini_ecommerce_appln


# Initialize a new git repository
git init

# Stage all files for the initial commit
git add .  

# Commit the staged files with a message

# Rename the default branch to main
git branch -M main

# Add the remote repository URL
git remote add origin https://github.com/mugabiBenjamin/mini_ecommerce_appln.git

git push -u origin main
```
